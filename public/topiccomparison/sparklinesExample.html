        height = 100;
        width = 400;
        data = [13, 15, 23, ... 9, 20, 10];

        maxDataValue = Math.max.apply(null, data);
        minDataValue = Math.min.apply(null, data);

        yScale = d3.scale.linear()
            .domain([minDataValue, maxDataValue])
            .range([height, 0]);

        xIncrement = width / (data.length - 1);
        
        lineGenerator = d3.svg.line()
            .x(function(d, i) {
                return (i * xIncrement); //moves the x position uniformly
            })
            .y(function(d) {
                return yScale(d); //use data values for y position
            });

        pathDataString = lineGenerator(data);

        d3.select('svg')
            .attr('width', width)
            .attr('height', height)
            .append('path')
            .style('stroke', '#76b76c')
            .attr('d', pathDataString);
